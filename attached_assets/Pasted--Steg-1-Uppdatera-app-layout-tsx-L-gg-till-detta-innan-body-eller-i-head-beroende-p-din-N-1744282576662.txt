// Inuti layout.tsx eller _document.tsx (för äldre Next.js)
'use client';

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <head>
        {/* ElevenLabs script direkt i head */}
        <script
          src="https://elevenlabs.io/convai-widget/index.js"
          async
          defer
          crossOrigin="anonymous"
        ></script>
      </head>
      <body>{children}</body>
    </html>
  );
}
